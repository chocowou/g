#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int size_i=board.size();
    int size_j=board[0].size();
    for(int i=0; i<size_i; i++){
        for(int j=0; j<size_j; j++){
            if(i+j<=k)answer+=board[i][j];
            else{break;}
        }
    }
    return answer;
}
